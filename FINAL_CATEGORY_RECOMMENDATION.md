```javascript
    // Search functionality
    function handleSearch(query) {
      const searchResults = document.getElementById('searchResults');
      const input = query.toLowerCase().trim();
      
      if (input.length < 2) {
        searchResults.classList.remove('active');
        return;
      }
      
      const matches = window.PRODUCTS.filter(p => {
        const name = p.name.toLowerCase();
        const desc = p.description.toLowerCase();
        const tags = p.seoTags || [];
        
        return name.includes(input) || 
               desc.includes(input) || 
               tags.some(tag => tag.includes(input));
      });
      
      if (matches.length === 0) {
        searchResults.innerHTML = '<div class="search-no-results">No items found for "' + query + '"</div>';
        searchResults.classList.add('active');
        return;
      }
      
      // Group by tag matches
      const tagGroups = {};
      matches.forEach(p => {
        const matchingTags = (p.seoTags || []).filter(tag => tag.includes(input));
        matchingTags.forEach(tag => {
          if (!tagGroups[tag]) tagGroups[tag] = [];
          tagGroups[tag].push(p);
        });
      });
      
      let html = '';
      
      // Show tag groups first
      Object.keys(tagGroups).slice(0, 5).forEach(tag => {
        const count = tagGroups[tag].length;
        html += `
          <div class="search-result-item" onclick="filterByTag('${tag}')">
            <span class="search-tag-match">${tag}</span> → ${count} item${count > 1 ? 's' : ''}
          </div>
        `;
      });
      
      // Then show individual products
      matches.slice(0, 8).forEach(p => {
        html += `
          <div class="search-result-item" onclick="scrollToProduct('${p.lot}')">
            ${p.name} <span style="color: #ffcc00;">$${p.price}</span>
          </div>
        `;
      });
      
      searchResults.innerHTML = html;
      searchResults.classList.add('active');
    }
    
    function filterByTag(tag) {
      const cards = document.querySelectorAll('.lot-card');
      let foundCount = 0;
      
      cards.forEach(card => {
        const indexStr = card.getAttribute('data-lot-index');
        const index = parseInt(indexStr);
        const product = window.PRODUCTS[index];
        const tags = product.seoTags || [];
        
        if (tags.includes(tag)) {
          card.classList.remove('hidden');
          foundCount++;
        } else {
          card.classList.add('hidden');
        }
      });
      
      // Close search
      document.getElementById('searchResults').classList.remove('active');
      document.getElementById('searchInput').value = '';
      
      // Scroll to collection
      document.getElementById('collection').scrollIntoView({ behavior: 'smooth' });
      
      // Update button text
      document.getElementById('categoryBtn').innerHTML = 
        `⚡ ${tag} (${foundCount}) <span>▼</span>`;
    }
    
    function scrollToProduct(lotId) {
      const cards = document.querySelectorAll('.lot-card');
      cards.forEach(card => {
        const indexStr = card.getAttribute('data-lot-index');
        const index = parseInt(indexStr);
        const product = window.PRODUCTS[index];
        
        if (product.lot === lotId) {
          card.scrollIntoView({ behavior: 'smooth', block: 'center' });
          card.style.animation = 'pulse 1s ease-in-out';
          setTimeout(() => card.style.animation = '', 1000);
        }
      });
      
      document.getElementById('searchResults').classList.remove('active');
      document.getElementById('searchInput').value = '';
    }
    
    // Close search results when clicking outside
    document.addEventListener('click', (e) => {
      const searchContainer = document.querySelector('.search-container');
      if (searchContainer && !searchContainer.contains(e.target)) {
        document.getElementById('searchResults').classList.remove('active');
      }
    });
```
